---
title: List Messages
description: Fetch feed posts for the current user
---

```typescript
import { whopApi } from "@/lib/whop-api";

const result = await whopApi.listMessages({
	// The type of feed to fetch from
	type: "chat_feed" /* Valid values: chat_feed | dms_feed | forum_feed | livestream_feed | universal_post | user */ /* Required! */,

	// The ID of the feed to fetch posts from
	id: "xxxxxxxxxxx" /* Required! */,
});

```

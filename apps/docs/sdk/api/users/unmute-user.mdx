<Note>This operation is only available on the server.</Note>
```typescript
import { whopSdk } from "@/lib/whop-sdk";

const result = await whopSdk.users.unmuteUser({
	// The ID of the user to unmute
	userId: "user_XXXXXXXX" /* Required! */,
});
```
